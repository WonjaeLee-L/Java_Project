1. member
create table member(
name varchar2(10) primary key,
id varchar2(10) not null unique,
password varchar2(16) not null,
id_num varchar2(13) not null unique,
cer_name_1 varchar2(10) default '미등록',
cer_name_2 varchar2(10) default '미등록',
constraint m_cer_name foreign key(cer_name_1) REFERENCES certificate(cer_name) on delete set null,
constraint m_cer_name2 foreign key(cer_name_2) REFERENCES certificate(cer_name) on delete set null);
// 인정되지 않는 자격증은 삭제되어 회원 정보에서도 삭제

2. counsel
create table counsel(
name varchar2(10),
interest varchar2(10) not null,
ay_date timestamp default sysdate not null,
memo varchar2(50),
recommend varchar2(10),
cs_date timestamp default sysdate,
score number(3),
attitude number(3) default 100,
constraint m_cs_name foreign key(name) REFERENCES member(name) on delete cascade);
// 회원 삭제 >> 상담 정보 삭제

##. apply
create table apply(
name varchar2(10) foreign key,
a_date timestamp default sysdate,
accept varchar2(1) not null check (accept in('Y','N')),
ac_date timestamp default sysdate,
constraint m_a_name foreign key(name) member(name) on delete cascade);
